    Class MockerTestBase<TComponent>

    Mock Test Base with Automatic Mocking using Mocker. This class contains the Mocks property to create and track all Mocks from creation of the component.

    Example test class using MockerTestBase<Car>.

    public class CarTests : MockerTestBase<Car>
{
    [Fact]
    public void TestCar()
    {
        Component.CarService.Should().NotBeNull();
    }
}

    ICarService is automatically injected into the Car object when the base class creates the component.

    public partial class Car
{
    public Car(ICarService carService) => CarService = carService;
}

    To configure mocks before the component is created, override SetupMocksAction.

    protected override Action<Mocker> SetupMocksAction => mocker =>
{
    mocker.GetMockDbContext<ApplicationDbContext>();
};

    Provide explicit invalid and valid values when constructor parameter defaults need to vary by type.

    [Fact]
public void Service_NullArgChecks_WithOverrides() => TestConstructorParameters(
    (action, constructorName, parameterName) =>
    {
        action
            .Should()
            .Throw<ArgumentNullException>()
            .WithMessage($"*{parameterName}*");
    },
    info => info.ParameterType.Name switch
    {
        "string" => string.Empty,
        "int" => -1,
        _ => default,
    },
